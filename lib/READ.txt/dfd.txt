class _MainpageState extends State<Mainpage> {
  int currentIndex = 0;
  final screens = [
    const firstpage(),
    const secondpage(),
    const thirdpage(),
    const fourthpage()
  ];
  @override
  Widget build(BuildContext context) => MaterialApp(
        initialRoute: '/1',
        routes: {'/1': (context) => const loading(),
        '/2': (context) => const firstpage()
        },
        debugShowCheckedModeBanner: false,
        home: Scaffold(
          body: screens[currentIndex],
          // ignore: prefer_const_literals_to_create_immutables
          bottomNavigationBar: BottomNavigationBar(
              iconSize: 25,
              // type: BottomNavigationBarType.fixed, it helps to fix the animation
              onTap: (index) => setState(() => currentIndex =
                  index), //setstate is a function that updates the UI immediately
              currentIndex: currentIndex,
              items: const [
                BottomNavigationBarItem(
                    icon: Icon(Icons.home),
                    label: "First",
                    backgroundColor: Color.fromARGB(0, 0, 0, 0)),

                    icon: Icon(Icons.message),
                    label: "Second",

                    label: "Third",
                    backgroundColor: Colors.pinkAccent),
                BottomNavigationBarItem(
                    icon: Icon(Icons.settings),
                    label: "Fourth",
                    backgroundColor: Colors.deepPurple)
              ]),
        ),
      );
}
